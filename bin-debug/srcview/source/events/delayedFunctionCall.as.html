package events
{
    import flash.events.TimerEvent;
    import flash.utils.Timer;
    
    
    
    public class delayedFunctionCall
    {
        private var myTimer:Timer;
        private var myTimerEvent:TimerEvent;
        private var delayedFunction:Function;
        
        public function delayedFunctionCall(whatFunctionToDelay:Function, delayInMilliseconds:Number)
        {
            this.delayedFunction = whatFunctionToDelay;
            myTimer = new Timer(delayInMilliseconds, 1);
            myTimer.addEventListener(TimerEvent.TIMER_COMPLETE, callDelayedFunction);
            myTimer.start();
        }
        
        private function callDelayedFunction(event:TimerEvent) {
            myTimer.removeEventListener(TimerEvent.TIMER_COMPLETE, callDelayedFunction);
            //trace("delayyedFunction: " + this.delayedFunction);
            delayedFunction();
        }
    }
}
